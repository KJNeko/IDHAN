69constexpr std::uint8_t decodeHexCharacters( const char left, const char right )
70{

81{
82 std::vector< std::byte > result {};

84
85 for ( std::size_t i = 0; i < str.size(); i += 2 )
86 {
87 const char first { str[ i ] };
