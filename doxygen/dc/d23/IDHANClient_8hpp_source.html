256template < typename TPromise >
257auto defaultErrorHandler( TPromise&& promise )
258{
259 auto handler =
260 [ promise ]( QNetworkReply* reply, [[maybe_unused]] QNetworkReply::NetworkError error, std::string server_msg )
261 {
262 // logging::logResponse( reply );
263
264 const std::runtime_error exception { format_ns::format( "{}", server_msg ) };
265
266 logging::error( server_msg );
267
268 promise->setException( std::make_exception_ptr( exception ) );
269
270 promise->finish();
271 reply->deleteLater();
272 };
273
274 return handler;
275}
