47template < typename T >
48std::optional< T > tryGetEnv( const std::string_view group, const std::string_view name )
49{
50 auto upper_group { std::string( group ) };
51 std::transform( upper_group.begin(), upper_group.end(), upper_group.begin(), ::toupper );
52 auto upper_name { std::string( name ) };
53 std::transform( upper_name.begin(), upper_name.end(), upper_name.begin(), ::toupper );
54
55 const auto env_name { std::format( "IDHAN_{}_{}", upper_group, upper_name ) };
56
57 if ( const char* value = std::getenv( env_name.data() ); value )
58 {
59 log::info( "Loaded config from env: {}={}", env_name, value );
60 if constexpr ( std::is_same_v< T, std::string > )
61 {
62 return std::string( value );
63 }
64 else if constexpr ( std::is_integral_v< T > )
65 {
66 return std::optional< T >( std::stoll( value ) );
67 }
68 else
69 {
70 static_assert( false, "Invalid type for ENV" );
71 }
72 }
73
74 return std::nullopt;
75}

100template < typename T >
101std::optional< T > getValue( const std::string_view path, const std::string_view group, const std::string_view name )
102{
103 if ( !std::filesystem::exists( path ) ) return std::nullopt;
104
105 try
106 {
107 auto config = toml::parse_file( path.data() );
108 if ( auto* table = config[ group ].as_table() )
109 {
110 if ( const auto value = ( *table )[ name ] )
111 {
112 if constexpr (
113 std::is_same_v< T, std::string > || std::is_same_v< T, bool > || std::is_same_v< T, double >
114 || std::is_same_v< T, std::int64_t > )
115 {
116 return **value.as< T >();
117 }
118 else if constexpr ( std::is_integral_v< T > )
119 {
120 return static_cast< T >( **value.as< std::int64_t >() );
121 }
122 else
123 static_assert( false, "Unsupported toml config type" );
124 }
125 return std::nullopt;
126 }
127 return std::nullopt;
128 }
129 catch ( const toml::parse_error& err )
130 {
131 log::warn( "Failed to parse config file: {}", err.what() );
132 }
133
134 return std::nullopt;
135}

137template < typename T >
138std::optional< T > getValue( const std::string_view group, const std::string_view name )
139{
140 //TODO: Get arguments from CLI
141
142 // ENV
143 if ( auto result = tryGetEnv< T >( group, name ); result ) return *result;
144
145 // overriden config path
146 const auto user_config_path { getUserConfigPath() };
147 if ( user_config_path.empty() )
148 {
149 if ( auto result = getValue< T >( user_config_path, group, name ); result )
150 {
151 return *result;
152 }
153 }
154
155 // priority paths
156 for ( const auto& path : config_paths | std::views::reverse )
157 {
158 if ( auto result = getValue< T >( path, group, name ); result )
159 {
160 return *result;
161 }
162 }
163
164 return std::nullopt;
165}
