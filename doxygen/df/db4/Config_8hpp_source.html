49template < typename T >
50std::optional< T > tryGetEnv( const std::string_view group, const std::string_view name )
51{
52 auto upper_group { std::string( group ) };
53 std::transform( upper_group.begin(), upper_group.end(), upper_group.begin(), ::toupper );
54 auto upper_name { std::string( name ) };
55 std::transform( upper_name.begin(), upper_name.end(), upper_name.begin(), ::toupper );
56
57 const auto env_name { std::format( "IDHAN_{}_{}", upper_group, upper_name ) };
58
59 if ( const char* value = std::getenv( env_name.data() ); value )
60 {
61 log::info( "Loaded config from env: {}={}", env_name, value );
62 if constexpr ( std::is_same_v< T, std::string > )
63 {
64 return std::string( value );
65 }
66 else if constexpr ( std::is_integral_v< T > )
67 {
68 return std::optional< T >( std::stoll( value ) );
69 }
70 else
71 {
72 static_assert( false, "Invalid type for ENV" );
73 }
74 }
75
76 return std::nullopt;
77}

102inline std::filesystem::path expand_home( std::string_view path )
103{
104 if ( !path.starts_with( "~" ) ) return path;
105
106 const char* home_path { std::getenv( "HOME" ) };
107
108 if ( !home_path )
109 {
110 log::warn( "Unable to get home path from ENV" );
111 return { path };
112 }
113
114 const std::filesystem::path home { home_path };
115 return home / path.substr( 2 );
116}

118template < typename T >
119std::optional< T > getValueFromFile(
120 const std::string_view path,
121 const std::string_view group,
122 const std::string_view name )
123{
124 const std::filesystem::path p { expand_home( path ) };
125
126 if ( !std::filesystem::exists( p ) ) return std::nullopt;
127
128 try
129 {
130 auto config = toml::parse_file( p.string() );
131 if ( auto* table = config[ group ].as_table() )
132 {
133 if ( const auto value = ( *table )[ name ] )
134 {
135 if constexpr (
136 std::is_same_v< T, bool > || std::is_same_v< T, double > || std::is_same_v< T, std::int64_t > )
137 {
138 return **value.as< T >();
139 }
140 else if constexpr ( std::is_same_v< T, std::string > )
141 {
142 return **value.as_string();
143 }
144 else if constexpr ( std::is_integral_v< T > )
145 {
146 return static_cast< T >( **value.as< std::int64_t >() );
147 }
148 else
149 static_assert( false, "Unsupported toml config type" );
150 }
151 return std::nullopt;
152 }
153 return std::nullopt;
154 }
155 catch ( const toml::parse_error& err )
156 {
157 log::warn( "Failed to parse config file: {}", err.what() );
158 }
159
160 return std::nullopt;
161}

163template < typename T >
164std::optional< T > getValue( const std::string_view group, const std::string_view name )
165{
166 // TODO: Get arguments from CLI
167
168 // ENV
169 if ( auto result = tryGetEnv< T >( group, name ); result ) return *result;
170
171 // overriden config path
172 const auto user_config_path { getUserConfigPath() };
173 if ( user_config_path.empty() )
174 {
175 if ( auto result = getValueFromFile< T >( user_config_path, group, name ); result )
176 {
177 return *result;
178 }
179 }
180
181 // priority paths
182 for ( const auto& path : config_paths | std::views::reverse )
183 {
184 if ( auto result = getValueFromFile< T >( path, group, name ); result )
185 {
186 log::info( "Loaded config from {}: {}.{}={}", path, group, name, *result );
187 return *result;
188 }
189 }
190
191 return std::nullopt;
192}
