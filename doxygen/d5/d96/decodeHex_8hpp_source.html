69constexpr std::uint8_t decodeHexCharacters( const char left, const char right )
70{
71 const std::uint8_t left_char { static_cast< std::uint8_t >( decodeHexCharacter( left ) << 4 ) };
72 const std::uint8_t right_char { decodeHexCharacter( right ) };
73
74 const std::uint8_t result { static_cast< std::uint8_t >( left_char | right_char ) };
75
